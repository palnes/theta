# Primitive Color Palette

The foundation color values. These raw colors form the basis of our semantic color system.

> **Note**: We recommend using semantic colors (`sys.color.*`) in your components instead of these primitives. Primitives are best referenced when creating new semantic tokens.

export const ColorScale = ({ name, colors }) => {
  return (
    <div style={{ marginBottom: 48 }}>
      <h2 style={{ textTransform: 'capitalize' }}>{name} Scale</h2>
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(120px, 1fr))', gap: 16 }}>
        {colors.map(({ shade, value, cssVariable, jsPath }) => {
          return (
            <div key={shade}>
              <div
                style={{
                  height: 60,
                  backgroundColor: `var(--ref-color-${name.toLowerCase()}-${shade})`,
                  borderRadius: 8,
                  marginBottom: 8,
                  border: '1px solid rgba(0, 0, 0, 0.1)',
                }}
              />
              <div style={{ fontSize: 11 }}>
                <strong>{name} {shade}</strong>
                <code style={{ display: 'block', fontSize: 10, color: '#666', marginTop: 2 }}>
                  {cssVariable}
                </code>
                <code style={{ display: 'block', fontSize: 9, color: '#999', marginTop: 2 }}>
                  {jsPath.replace('tokens.', '')}
                </code>
                <div style={{ fontSize: 9, color: '#888', marginTop: 2, fontFamily: 'monospace' }}>
                  {value}
                </div>
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
};

export const SpecialColors = () => {
  const { data, loading, error } = useDocumentationData();
  
  if (loading) return <div>Loading special colors...</div>;
  if (error) return <div>Error loading colors: {error}</div>;
  if (!data) return null;
  
  const specials = getSpecialColorsFromDocs(data.ref?.color || []);
  
  return (
    <div style={{ marginBottom: 48 }}>
      <h2>Special Colors</h2>
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(120px, 1fr))', gap: 16 }}>
        {specials.map(({ name, token, cssVariable, jsPath }) => {
          
          return (
            <div key={name}>
              <div
                style={{
                  height: 60,
                  backgroundColor: name === 'transparent' ? 'transparent' : `var(--ref-color-${name})`,
                  borderRadius: 8,
                  marginBottom: 8,
                  border: '1px solid rgba(0, 0, 0, 0.1)',
                  background: name === 'transparent' 
                    ? 'repeating-linear-gradient(45deg, #f0f0f0, #f0f0f0 10px, #fff 10px, #fff 20px)' 
                    : `var(--ref-color-${name})`,
                }}
              />
              <div style={{ fontSize: 11 }}>
                <strong style={{ textTransform: 'capitalize' }}>{name}</strong>
                <code style={{ display: 'block', fontSize: 10, color: '#666', marginTop: 2 }}>
                  {cssVariable}
                </code>
                <code style={{ display: 'block', fontSize: 9, color: '#999', marginTop: 2 }}>
                  {jsPath.replace('tokens.', '')}
                </code>
                <div style={{ fontSize: 9, color: '#888', marginTop: 2, fontFamily: 'monospace' }}>
                  {token.$value}
                </div>
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
};

export const AllColorScales = () => {
  const { data, loading, error } = useDocumentationData();
  
  if (loading) return <div>Loading color scales...</div>;
  if (error) return <div>Error loading colors: {error}</div>;
  if (!data) return null;
  
  const colorScales = getColorScalesFromDocs(data.ref?.color || []);
  
  return (
    <>
      {colorScales.map(({ name, colors }) => (
        <ColorScale key={name} name={name} colors={colors} />
      ))}
    </>
  );
};
